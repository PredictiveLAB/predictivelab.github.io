[docs]    def init_arch(self):
        """Initializes the Random Forest model.
        Introduces the following values in arch_params:
            - arch_params["n_estimators"] (int): The number of trees in the forest.
                Range is (1, inf).
            - arch_params["max_depth"] (int): The maximum depth of the tree.
                Range is (1, inf).
            - arch_params["min_samples_split"] (int): The minimum number of samples required to split an internal node.
                Range is (2, inf).
        """
        self.model = RandomForestClassifier(
            n_estimators=self.arch_params["n_estimators"],
            max_depth=self.arch_params["max_depth"],
            min_samples_split=self.arch_params["min_samples_split"],
        )

        if self.chckpt_filepath is not None:
            self.load_model()

        return self

[docs]    def init_arch(self):
        """Initialize the model architecture.

        Introduces the following necessary values in arch_params:
            - arch_params["feature_size"] (int): The number of input features.
            - arch_params["h_num"] (int): The number of hidden layers.
            - arch_params["h_size"] (int): The number of units in each hidden layer.
            - arch_params["out_size"] (int): The number of output classes.
            - arch_params["metric"] (str): The metric to use.
        """
        feature_size = self.arch_params["feature_size"]
        h_num = self.arch_params["h_num"]
        h_size = self.arch_params["h_size"]
        out_size = self.arch_params["out_size"]
        metric = self.arch_params.get("metric")

        # define output activation function
        if out_size > 2:
            out_activation = "softmax"
            loss = "sparse_categorical_crossentropy"
        else:
            # if out_size == 2, change it to 1, since it's a binary problem
            out_size = 1
            out_activation = "sigmoid"
            loss = "binary_crossentropy"

        layers_list = [
            {
                "layer_type": "input_layer",
                "input_shape": (feature_size,),
            }
        ]
        for _ in range(h_num):
            layers_list.append({"layer_type": "dense", "units": h_size, "activation": "relu"})
        layers_list.append({"layer_type": "dense", "units": out_size, "activation": out_activation})
        compile_kwargs = {"optimizer": "adam", "loss": loss}
        if metric is not None:
            compile_kwargs.update({"metrics": [metric]})
        self.model = keras_wrapper.KerasNet(layers_list, compile_kwargs, self.chckpt_filepath)

        return self

        - train_params["epochs"] (int): Number of epochs for training
    """

    def __init__(self, arch_params, chckpt_filepath=None) -> None:
        super(CNN, self).__init__(arch_params, chckpt_filepath)
        self.use_split_windows = True
        self.window_size = self.arch_params["time_size"]
        self.overlap = self.arch_params["overlap"]


[docs]    def init_arch(self):
        """Initialize the model architecture.

        Introduces the following necessary values in arch_params:
            - arch_params["time_size"] (int): Number of time steps in the input to use in each window.
            - arch_params["feature_size"] (int): number of input features.
            - arch_params["filter_num"] (int): number of filters in the first CNN block (it is doubled in each block)
            - arch_params["kernel_v"] (int): vertical kernel size
            - arch_params["kernel_h"] (int): horizontal kernel size
            - arch_params["out_size"] (int): The number of output classes.
            - arch_params["metric"] (str): The metric to use.
        """
        v_size = self.arch_params["time_size"]
        h_size = self.arch_params["feature_size"]
        filter_num = self.arch_params["filter_num"]
        kernel_v = self.arch_params["kernel_v"]
        kernel_h = self.arch_params["kernel_h"]
        out_size = self.arch_params["out_size"]
        metric = self.arch_params.get("metric")

        # define output activation function
        if out_size > 2:
            out_activation = "softmax"
            loss = "sparse_categorical_crossentropy"
        else:
            # if out_size == 2, change it to 1, since it's a binary problem
            out_size = 1
            out_activation = "sigmoid"
            loss = "binary_crossentropy"

        # calculate how many times the input can be divided by 2
        block_num = int(np.min(np.log2([v_size, h_size])))

        layers_list = [
            {
                "layer_type": "input_layer",
                "input_shape": (v_size, h_size, 1),
            }
        ]
        for i in range(block_num):
            layers_list.append(
                {
                    "layer_type": "conv_block",
                    "filters": filter_num * 2 ** i,
                    "kernel_size": (kernel_v, kernel_h),
                    "downsample": True,
                }
            )
        layers_list.append({"layer_type": "global_avg_pooling"})
        layers_list.append({"layer_type": "flatten"})
        layers_list.append(
            {
                "layer_type": "dense",
                "units": out_size,
                "activation": out_activation,
            }
        )
        compile_kwargs = {"optimizer": "adam", "loss": loss}
        if metric is not None:
            compile_kwargs.update({"metrics": [metric]})
        self.model = keras_wrapper.KerasNet(layers_list, compile_kwargs, self.chckpt_filepath)

        return self

        """Initialize the model architecture.

            - arch_params["feature_size"] (int): The number of input features.
