[docs]class AutoMLP(AutoEncoderModel):
    """Autoencoder model based on a MLP architecture.
    Inherits from base_arch.AutoEncoderModel.

    AutoMLP is a model that uses a multilayer perceptron (MLP) architecture to encode
    and decode data, using the reconstruction error to output a KPI between 0 and 1,
    indicating how anomalous the data is. Uses the input data as a batch of vectors,
    where each vector is an instant of time with features, and feeds it to Dense layers.

    Redefines the methods:
        - init_arch: initializes the model architecture
        - parse_fit_logging:
        - parse_predict_logging:

    Requires the following values in arch_params:
        - arch_params["slope_coef"] (float): Slope (or scale) factor for KPI calculation,
            defines how quickly the KPI grows
        - arch_params["contamination"] (float): Proportion of outliers in the data set.
        - arch_params["feature_size"] (int): The number of input features.
        - arch_params["lat_size"] (int): The number of units in the latent layer
        - arch_params["h_num"] (int): Number of hidden layers

    Requires the following values in train_params (for fit method):
        - train_params["epochs"] (int): Number of epochs to train the model
        - train_params["batch_size"] (int): Batch size for training the model
    """

    def __init__(self, arch_params, chckpt_filepath=None) -> None:
        super().__init__(arch_params, chckpt_filepath)
        self.use_split_windows = False


[docs]    def init_arch(self):
        """Initializes the AutoMLP model architecture.

        Introduces the following necessary values in arch_params:
            - feature_size (int): The number of input features.
            - lat_size (int): The number of units in the latent layer
            - h_num (int): number of hidden layers
        """
        feature_size = self.arch_params["feature_size"]
        lat_size = self.arch_params["lat_size"]
        h_num = self.arch_params["h_num"]

        mid_sizes = np.linspace(start=feature_size, stop=lat_size, num=h_num + 2)
        mid_sizes = mid_sizes.astype(int)[1:-1]

        # input
        layers_list = [
            {
                "layer_type": "input_layer",
                "input_shape": (feature_size,),
            }
        ]
        # encoder without latent layer
        for layer_size in mid_sizes:
            layers_list.append({"layer_type": "dense", "units": layer_size, "activation": "relu"})
        # latent layer
        layers_list.append({"layer_type": "dense", "units": lat_size, "activation": "relu"})
        # decoder without output
        for layer_size in np.flip(mid_sizes):
            layers_list.append({"layer_type": "dense", "units": layer_size, "activation": "relu"})
        # output
        layers_list.append({"layer_type": "dense", "units": feature_size})

        compile_kwargs = {"optimizer": "adam", "loss": "mean_squared_error"}
        self.model = keras_wrapper.KerasNet(layers_list, compile_kwargs, self.chckpt_filepath)

        if self.chckpt_filepath is not None:
            self._load_scale_params(self.chckpt_filepath)

        return super(AutoMLP, self).parse_fit_logging(log_filepath)

    """Autoencoder model based on a CNN architecture.
    Inherits from base_arch.AutoEncoderModel.

    AutoCNN is a model that uses a convolutional neural network (CNN) architecture to encode
    and decode data, using the reconstruction error to output a KPI between 0 and 1,
    indicating how anomalous the data is. Transforms the input data into a batch of matrices (windows),
    where each matrix is time-window with features, and feeds it to Convolutional layers.

    Defines the method:
        - init_arch: initializes the model architecture
        - parse_fit_logging:
        - parse_predict_logging:

    Requires the following values in arch_params:
        - arch_params["slope_coef"] (float): Slope (or scale) factor for KPI calculation,
            defines how quickly the KPI grows
        - arch_params["contamination"] (float): Proportion of outliers in the data set.
        - arch_params["time_size"] (int): Number of time steps in the input to use in each window.
        - arch_params["overlap"] (int): Number of time steps to overlap between windows.
        - arch_params["feature_size"] (int): Number of features in the input.
        - arch_params["block_num"] (int): number of hidden CNN blocks in encoder and decoder
        - arch_params["filter_num"] (int): number of filters in the first CNN block (it is doubled in each block)
        - arch_params["kernel_v"] (int): vertical size of the convolution kernel
        - arch_params["kernel_h"] (int): horizontal size of the convolution kernel

    Requires the following values in train_params (for fit method):
        - train_params["epochs"] (int): Number of epochs to train the model
        - train_params["batch_size"] (int): Batch size for training the model
    """

    def __init__(self, arch_params, chckpt_filepath=None) -> None:
        super().__init__(arch_params, chckpt_filepath)
        self.use_split_windows = True
        self.window_size = self.arch_params["time_size"]
        self.overlap = self.arch_params["overlap"]


[docs]    def init_arch(self):
        """Initializes the AutoCNN model architecture.

        Introduces the following necessary values in arch_params:
            - time_size (int): vertical size of the input data
            - feature_size (int): horizontal size of the input data
            - block_num (int): number of hidden CNN blocks in encoder and decoder
            - filter_num (int): number of filters in the first CNN block (it is doubled in each block)
            - kernel_v (int): vertical size of the convolution kernel
            - kernel_h (int): horizontal size of the convolution kernel
        """
        v_size = self.arch_params["time_size"]
        h_size = self.arch_params["feature_size"]
        block_num = self.arch_params["block_num"]
        filter_num = self.arch_params["filter_num"]
        kernel_v = self.arch_params["kernel_v"]
        kernel_h = self.arch_params["kernel_h"]

        # power of 2 closer to h_size
        inter_h_size = int(2 ** np.round(np.log2(h_size)))

        # change h size to power of 2 (bool)
        change_h_size = h_size != inter_h_size

        # input
        layers_list = [
            {
                "layer_type": "input_layer",
                "input_shape": (v_size, h_size),
            }
        ]
        if change_h_size:
            # transform h_size to a power of 2
            layers_list.append({"layer_type": "dense", "units": inter_h_size, "activation": "relu"})

        # reshape to image format (3 axis)
        layers_list.append({"layer_type": "reshape", "target_shape": (v_size, inter_h_size, 1)})

        # encoder without latent layer
        for i in range(block_num):
            layers_list.append(
                {
                    "layer_type": "conv_block",
                    "filters": filter_num * 2 ** i,
                    "kernel_size": (kernel_v, kernel_h),
                    "downsample": True,
                }
            )
        # latent layer
        layers_list.append(
            {
                "layer_type": "conv_block",
                "filters": filter_num * 2 ** block_num,
                "kernel_size": (kernel_v, kernel_h),
                "downsample": True,
            }
        )
        # decoder without output
        for i in range(block_num):
            layers_list.append(
                {
                    "layer_type": "rev_conv_block",
                    "filters": filter_num * 2 ** (block_num - i - 1),
                    "kernel_size": (kernel_v, kernel_h),
                    "upsample": True,
                }
            )

        if change_h_size:
            # last conv before output
            layers_list.append(
                {
                    "layer_type": "rev_conv_block",
                    "filters": 1,
                    "kernel_size": (kernel_v, kernel_h),
                    "upsample": True,
                }
            )

            # reshape to original format (2 axis)
            layers_list.append({"layer_type": "reshape", "target_shape": (v_size, inter_h_size)})

            # transform power of 2 to h_size (output)
            layers_list.append({"layer_type": "dense", "units": h_size, "activation": "linear"})

        else:
            # last conv which is the output
            # add last_layer argument to the last conv block
            layers_list.append(
                {
                    "layer_type": "rev_conv_block",
                    "filters": 1,
                    "kernel_size": (kernel_v, kernel_h),
                    "upsample": True,
                    "last_layer": True,
                }
            )

            # reshape to original format (2 axis)
            layers_list.append({"layer_type": "reshape", "target_shape": (v_size, h_size)})

        compile_kwargs = {"optimizer": "adam", "loss": "mean_squared_error"}
        self.model = keras_wrapper.KerasNet(layers_list, compile_kwargs, self.chckpt_filepath)

        if self.chckpt_filepath is not None:
            self._load_scale_params(self.chckpt_filepath)

        return super(AutoCNN, self).parse_fit_logging(log_filepath)

[docs]    def init_arch(self):
        """Initializes the AutoRNN model architecture.

        Introduces the following necessary values in arch_params:
            - feature_size (int): The number of input features.
            - lat_size (int): The number of units in the latent layer
        """
        feature_size = self.arch_params["feature_size"]
        lat_size = self.arch_params["lat_size"]

        # input
        layers_list = [{"layer_type": "input_layer", "input_shape": (None, feature_size)}]
        layers_list.append({"layer_type": "dense", "units": feature_size, "activation": "relu"})
        # latent layer
        layers_list.append({"layer_type": "lstm", "units": lat_size, "return_sequences": True})
        # output
        layers_list.append({"layer_type": "lstm", "units": feature_size, "return_sequences": True})
        layers_list.append(
            {
                "layer_type": "dense",
                "units": feature_size,
            }
        )

        compile_kwargs = {"optimizer": "adam", "loss": "mean_squared_error"}
        self.model = keras_wrapper.KerasNet(layers_list, compile_kwargs, self.chckpt_filepath)

        if self.chckpt_filepath is not None:
            self._load_scale_params(self.chckpt_filepath)
