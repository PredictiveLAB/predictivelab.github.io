[docs]def plot_data(
        data,
        sensors_list,
        n_cols=2,
        title=None,
        suptit_sz=12,
        ylabel_sz=9.5   
):
    n_rows = int(
        len(sensors_list)/n_cols
    )
    # Set plot figure dimensions
    heigth = 5.5*n_rows
    width = 20*n_cols
    fig_size = (width, heigth)
    # Create figure and axes
    fig, axes = plt.subplots(ncols=n_cols, nrows=n_rows, figsize=fig_size,
                             dpi=100, sharex=True)
    # Set figure title
    if isinstance(title, str):
        fig.suptitle(title, size=suptit_sz)
    # Iter over each sensor data
    for ax, sensor_name in zip(axes.flat, sensors_list):
        ax.plot(
            data.index,
            data[sensor_name],
            linewidth=3,
            alpha=0.6
        )
        # Figure text
        ax.set_ylabel(sensor_name, size=ylabel_sz)

    plt.tight_layout()
    plt.show()

    

[docs]def plt_kpi_sensors(
    data_kpi,
    sensors_list,
    fail_data=None,
    time_col="Timestamp",
    fail_label_col="label",
    start_fail_col="start",
    end_fail_col="end",
    fail_label=1,
    title=None,
    n_cols=2,
    density=100,
    n_cols_legend=2,
    suptit_sz=22,
    xlabel_sz=9.5,
    ylabel_sz=15    
):

    n_rows = int(
        len(sensors_list)/n_cols
    )
    
    data_plot = get_dataplot_dict(data_kpi)
    
    colors = {
        "Normal" : "#2e7813",
        "Alerta 1" :"#edc421",
        "Alerta 2" : "#c9671c",
        "Peligro" : "#ff1100",
        "Falla registrada" : "#6e514f",
        "Detencion programada" : "#25d9e6",
        "Detencion NO registrada" : "#e87bdb"
    }
    
    heigth = 5.5*n_rows
    width = 20*n_cols
    fig_size = (width, heigth)

    fig, axes = plt.subplots(ncols=n_cols, nrows=n_rows, figsize=fig_size,
                             dpi=100, sharex=True)
    
    if isinstance(title, str):
        fig.suptitle(title, size=suptit_sz)
    
    for ax, sensor_name in zip(axes.flat, sensors_list):
        
        # ITER OVER EACH HEALTH CLASS
        for health_class in data_plot:
            
            ax.scatter(
                data_plot[health_class].index,
                data_plot[health_class][sensor_name],
                c=colors[health_class],
                edgecolor=colors[health_class],
                linewidth=1,
                alpha=0.6
            )
            


        # ITER OVER EACH FAIL EVENT
        if fail_data.shape[0] > 0:
            
            y = [
                data_kpi[sensor_name].max(),
                data_kpi[sensor_name].min()
            ]
            
            for fail_idx in fail_data.index:
                dates_linspace = pd.date_range(
                    start=fail_data.loc[fail_idx, start_fail_col],
                    end=fail_data.loc[fail_idx, end_fail_col],
                    periods=density
                ).to_list()
                
                if fail_data.loc[fail_idx, fail_label_col] == fail_label:
                    color = colors["Falla registrada"]
                else:
                    color = colors["Detencion programada"]
                for date in dates_linspace:
                    ax.plot(
                        [date, date],
                        y,
                        c=color,
                        linewidth=2.5,
                        zorder=-1 #Plot "behind" the scatter
                    )    

        
        # FIGURE TEXT    
        ax.set_ylabel(sensor_name, size=ylabel_sz)
        if ax == axes.flat[0]:
            ax.legend(
                labels=colors.keys(),
                ncol=n_cols_legend
            )
            # Change handle colors
            for handle_name, handle in zip(
                    colors,
                    ax.get_legend().legendHandles
            ):
                handle.set_color(colors[handle_name])
        
        
    plt.tight_layout()
    plt.show()




[docs]def get_dataplot_dict(data_kpi):
    data_plot = {
        "Normal" : data_kpi[
            data_kpi["kpi_pred"] <= 0.1
        ],
        
        "Alerta 1" : data_kpi[
            (data_kpi["kpi_pred"] > 0.1) &
            (data_kpi["kpi_pred"] <= 0.35)
        ],
        
        "Alerta 2" : data_kpi[
            (data_kpi["kpi_pred"] > 0.35) &
            (data_kpi["kpi_pred"] <= 0.65)
        ],
        
        "Peligro" : data_kpi[
            data_kpi["kpi_pred"] > 0.65
        ]    
    }
    return data_plot
