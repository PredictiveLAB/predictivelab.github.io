[docs]    def analyze(self, X, y=None, y_labels=None):
        # Return the DataFrame with filtered information and deleting the transient values
        # X = self._convert(X)
        # data = X.to_numpy()
        data = X
        Fs = self.sampl_frec
        L = self.seg_len
        Fmax = self.max_frec
        l = self.overlap
        nb = self.bn
        df = Fs / L

        Nf = int(Fmax / df)  # numero de l√≠neas en los espectros
        N = len(data)  # numero de datos
        Nt = math.floor((N - l) / (L - l))  # total de segmentos
        En = np.zeros((Nt, nb))
        for i in range(1, Nt + 1):
            inicio = (i - 1) * L - (i - 1) * l + 1
            fin = i * L - (i - 1) * l

            Fn = fft(data[inicio:fin])[0:Nf] / (L / 2)

            Lb = int(Nf / nb)

            for k in range(1, nb + 1):
                inicio = Lb * (k - 1) + 1
                fin = k * Lb
                En[i - 1][k - 1] = np.mean(abs(Fn[inicio:fin]))

            # print(En)

        return pd.DataFrame(En.mean(axis=0)), En.std(axis=0), np.arange(nb)




[docs]class BandFourier(BasePreprocess):
    def __init__(self, preprocess_params, chckpt_filepath=None):
        super(BandFourier, self).__init__()

        self.sensors_mask = preprocess_params.get("sensors_mask")
        self.sampling_rate = preprocess_params.get("sampling_rate")
        self.window = preprocess_params.get("window", "hanning")
        self.overlap = preprocess_params.get("overlap")
        self.segment_len = preprocess_params.get("segment_len")
        self.n_bands = preprocess_params.get("n_bands")

        # default params
        if self.overlap is None:
            self.overlap = 0
        if self.n_bands is None:
            self.n_bands = 1

    def _window_dict(self):
        if self.window == "None":
            return lambda x: 1

        elif self.window == "hanning":
            return np.hanning

        elif self.window == "hamming":
            return np.hamming

        elif self.window == "blackman":
            return np.bartlett

        elif self.window == "bartlett":
            return np.bartlett

    def _windowed_fourier(self, X_in, y, L, Nt, overlap, nb):
        X = X_in.values
        window = self._window_dict()(L - 1)
        first_idx = X.shape[0] - (Nt * (L - overlap) + overlap)
        Pn = np.zeros((Nt, nb))
        y_out = np.empty((Nt,))
        index_out = np.empty((Nt,))
        for i in range(1, Nt + 1):
            inicio = (i - 1) * L - (i - 1) * overlap + 1 + first_idx
            fin = i * L - (i - 1) * overlap + first_idx

            Fn = fft(X[inicio:fin] * window)[1 : int(L / 2)] / (L / 2)
            Lb = int(L / 2 / nb)
            for k in range(1, nb + 1):
                inicio_k = Lb * (k - 1) + 1
                fin_k = k * Lb
                Pn[i - 1][k - 1] = np.mean(abs(Fn[inicio_k:fin_k]))
            index_out[i - 1] = X_in.index[inicio:fin][-1]
            if y is not None:
                y_out[i - 1] = np.argmax(np.bincount((y[inicio:fin]).astype(int)))

        return Pn, y_out, index_out

[docs]class SimpleFourier(BasePreprocess):
    def __init__(self, preprocess_params, chckpt_filepath=None):
        super(SimpleFourier, self).__init__()

        self.sensors_mask = preprocess_params.get("sensors_mask")
        self.sampling_rate = preprocess_params.get("sampling_rate", 5)
        self.window = preprocess_params.get("window", "hanning")
        self.overlap = preprocess_params.get("overlap", 5)
        self.segment_len = preprocess_params.get("segment_len", 10)

    def _window_dict(self):
        if self.window == "None":
            return lambda x: 1

        elif self.window == "hanning":
            return np.hanning

        elif self.window == "hamming":
            return np.hamming

        elif self.window == "blackman":
            return np.bartlett

        elif self.window == "bartlett":
            return np.bartlett

    def _windowed_fourier(self, X, L, Nt, overlap):

        window = self._window_dict()(L - 1)
        Pn = np.zeros((L // 2 - 1,))
        for i in range(1, Nt + 1):
            inicio = (i - 1) * L - (i - 1) * overlap + 1
            fin = i * L - (i - 1) * overlap

            Fn = fft(X[inicio:fin] * window)[1 : int(L / 2)] / (L / 2)
            Pn = Pn + abs(Fn) / Nt

        return Pn
