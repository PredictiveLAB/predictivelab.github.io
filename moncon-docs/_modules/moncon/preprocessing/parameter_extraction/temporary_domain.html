    def __init__(self, preprocess_params, chckpt_filepath=None):
        super(TemporaryDomain, self).__init__()

        self.sensors_mask = preprocess_params.get("sensors_mask")
        self.overlap = preprocess_params.get("overlap")  # overlap used
        self.segment = preprocess_params.get("segment")  # length of segment
        self.feat_names = preprocess_params.get("feats_names")

        # include derivative and integral
        self.use_dev_int = preprocess_params.get("use_dev_int")

        # default params
        if self.overlap is None:
            self.overlap = 0
        if self.feat_names is None:
            self.feat_names = [
                "RMS",
                "Peak",
                "Peak-Peak",
                "Crest",
                "Mean",
                "Var",
                "Asymmetry",
                "Kurtosis",
                "Moment_5",
            ]
        if self.use_dev_int is None:
            self.use_dev_int = False

        if self.use_dev_int:
            self.n_feats = 3 * len(self.feat_names)
            self.aggr_f = self.time_features
        else:
            self.n_feats = len(self.feat_names)
            self.aggr_f = self.features

        self.step = self.segment - self.overlap


[docs]    def features(self, x, axis=0):
        """
        Extract features
        """

        feats = []
        if "RMS" in self.feat_names:
            feats.append(np.sqrt(np.nanmean(np.square(x), axis=axis)))
        if "Peak" in self.feat_names:
            feats.append(np.amax(x, axis=axis))
        if "Peak-Peak" in self.feat_names:
            feats.append(np.amax(x, axis=axis) - np.amin(x, axis=axis))
        if "Crest" in self.feat_names:
            rms = np.sqrt(np.nanmean(np.square(x), axis=axis))
            peak = np.amax(x, axis=axis)
            feats.append(peak / (rms + 0.001))
        if "Mean" in self.feat_names:
            feats.append(np.nanmean(x, axis=axis))
        if "Var" in self.feat_names:
            feats.append(np.nanvar(x, axis=axis))
        if "Asymmetry" in self.feat_names:
            feats.append(skew(x, axis=axis, nan_policy="omit").data)
        if "Kurtosis" in self.feat_names:
            feats.append(kurtosis(x, axis=axis, nan_policy="omit").data)  # 4th moment
        if "Moment_5" in self.feat_names:
            feats.append(moment(x, 5, axis=axis, nan_policy="omit").data)

        return np.concatenate(feats, axis=0)



[docs]    def time_features(self, x, axis=0):
        """
        Calculate the features for each column, its integral and its derivatitive
        """

        dx = x[1:] - x[:-1]
        ix = x[1:] + x[:-1]

        fx = self.features(x)
        fdx = self.features(dx)
        fix = self.features(ix)

        return np.concatenate((fx, fdx, fix), axis=0)

    def __set_n_components(self, X):
        self.n_components = self.n_feats * X.shape[1]

    def __num_win(self, X):
        return (
            int(np.floor((X.shape[0] - self.overlap) / self.step))
            if X.shape[0] > self.overlap
            else 1
        )

[docs]    def fit_transform(self, X, y, out_chckpt_filepath):
        self.__set_n_components(X)
        num_win = self.__num_win(X)
        x_in = X.to_numpy()
        first_idx = x_in.shape[0] - (num_win * (self.step) + self.overlap)
        X_out = np.empty((num_win, self.n_components))
        y_out = np.empty((num_win,))
        index_out = np.empty((num_win,))
        for i in range(num_win):
            start = i * self.step + first_idx
            end = start + self.segment
            X_out[i] = self.aggr_f(x_in[start:end], axis=0)
            y_out[i] = np.argmax(np.bincount((y[start:end]).astype(int)))
            index_out[i] = X.index[start:end][-1]

        X_out = pd.DataFrame(X_out, index=index_out)
        y_out = pd.Series(y_out, index=index_out)
        self.set_n_components(X_out)
        return X_out, y_out

        self.__set_n_components(X)
        num_win = self.__num_win(X)
        X_out = np.empty((num_win, self.n_components))
        index_out = np.empty((num_win,))
        for i in range(num_win):
            start = i * self.step
            end = start + self.segment
            X_out[i] = self.aggr_f(X[start:end], axis=0)
            index_out[i] = X.index[start:end][-1]

        X_out = pd.DataFrame(X_out, index=index_out)
        self.set_n_components(X_out)
        return X_out

        box_plot_data = []

            # feat_data = [feat_data[:, j] for j in range(X.shape[1])]
            bp = plt.boxplot(feat_data)
            box_plot_data.append(TemporaryDomain.get_box_plot_data(bp, X.shape[1]))

        return self.feat_names, box_plot_data
