[docs]class CnnBlock(layers.Layer):
    def __init__(
        self, downsample: bool, filters: int, kernel_size: tuple = (3, 3), **kwargs
    ):
        super(CnnBlock, self).__init__(**kwargs)
        self.downsample = downsample

        self.conv1 = layers.Conv2D(filters, kernel_size=kernel_size, padding="same")
        self.bn1 = layers.BatchNormalization()
        self.act1 = layers.Activation("relu")
        self.conv2 = layers.Conv2D(filters, kernel_size=kernel_size, padding="same")
        self.bn2 = layers.BatchNormalization()
        self.act2 = layers.Activation("relu")
        self.maxp = layers.MaxPool2D(pool_size=(2, 2))


[docs]    def call(self, inputs):
        y = self.conv1(inputs)
        y = self.bn1(y)
        y = self.act1(y)
        y = self.conv2(y)
        y = self.bn2(y)
        y = self.act2(y)
        if self.downsample:
            y = self.maxp(y)
        return y




[docs]class ReverseCnnBlock(layers.Layer):
    def __init__(
        self,
        upsample: bool,
        filters: int,
        kernel_size: tuple = (3, 3),
        last_layer: bool = False,
        **kwargs,
    ):
        super(ReverseCnnBlock, self).__init__(**kwargs)
        self.upsample = upsample

        self.convt1 = layers.Conv2DTranspose(
            filters, kernel_size=kernel_size, padding="same"
        )
        self.bn1 = layers.BatchNormalization()
        self.act1 = layers.Activation("relu")
        self.convt2 = layers.Conv2DTranspose(
            filters, kernel_size=kernel_size, padding="same"
        )
        self.bn2 = layers.BatchNormalization()
        if last_layer:
            self.act2 = layers.Activation("linear")
        else:
            self.act2 = layers.Activation("relu")
        self.up = layers.UpSampling2D(size=(2, 2))


[docs]    def call(self, inputs):
        y = self.convt1(inputs)
        y = self.bn1(y)
        y = self.act1(y)
        y = self.convt2(y)
        y = self.bn2(y)
        y = self.act2(y)
        if self.upsample:
            y = self.up(y)
        return y




[docs]class KerasNet:
    keras_layer_map = {
        # core
        "input_layer": layers.InputLayer,
        "reshape": layers.Reshape,
        "batch_norm": layers.BatchNormalization,
        "dropout": layers.Dropout,
        "flatten": layers.Flatten,
        "activation": layers.Activation,
        "dense": layers.Dense,
        # cnn
        "conv": layers.Conv2D,
        "conv_transpose": layers.Conv2DTranspose,
        "avg_pooling": layers.AveragePooling2D,
        "global_avg_pooling": layers.GlobalAveragePooling2D,
        "max_pool": layers.MaxPool2D,
        "conv_block": CnnBlock,
        "rev_conv_block": ReverseCnnBlock,
        # rnn
        "simple_rnn": layers.SimpleRNN,
        "gru": layers.GRU,
        "lstm": layers.LSTM,
        "repeat_vector": layers.RepeatVector,
    }
    keras_compile_options = [
        "optimizer",
        "loss",
        "metrics",
        "loss_weights",
        "weighted_metrics",
    ]

    def __init__(
        self,
        layers_list: List[Dict[str, Any]],
        compile_kwargs: Dict[str, Any],
        chckpt_filepath: str = None,
    ) -> None:

        self.layers_list = copy.deepcopy(layers_list)
        self.compile_kwargs = copy.deepcopy(compile_kwargs)

        self.model = self.sequential_model_creator()
        if chckpt_filepath is not None:
            load_obj = self.model.load_weights(chckpt_filepath)
            if load_obj is not None:
                load_obj.expect_partial()



[docs]    def sequential_model_creator(self):
        model = Sequential()
        for layer_dict in self.layers_list:
            layer_class = self.keras_layer_map[layer_dict["layer_type"]]
            layer_kwargs = {i: layer_dict[i] for i in layer_dict if i != "layer_type"}
            model.add(layer_class(**layer_kwargs))

        model.compile(**self.compile_kwargs)
        return model



[docs]    def fit(
        self,
        x_train,
        y_train,
        epochs,
        batch_size,
        out_chckpt_filepath,
        log_filepath,
        x_val=None,
        y_val=None,
    ):

        if x_val is None or y_val is None:
            val_data = None
        else:
            val_data = (x_val, y_val)

        model_checkpoint_callback = callbacks.ModelCheckpoint(
            filepath=out_chckpt_filepath,
            monitor="loss" if val_data is None else "val_loss",
            save_weights_only=True,
            save_best_only=True,
        )

        if log_filepath is not None:
            self.scores = ["loss"]
            if val_data is not None:
                self.scores.append("val_loss")
            if "metrics" in self.compile_kwargs:
                metric = self.compile_kwargs["metrics"][0]
                self.scores.append(metric)
                if val_data is not None:
                    self.scores.append(f"val_{metric}")

            str_epoch = (
                lambda epoch, logs: f"epoch: {epoch}, "
                + ", ".join(f"{mtrc}: {logs[mtrc]:.4f}" for mtrc in self.scores)
                + "\n"
            )

            with open(log_filepath, mode="a", buffering=1) as txt_log:
                logging_callback = callbacks.LambdaCallback(
                    on_epoch_end=lambda epoch, logs: txt_log.write(
                        str_epoch(epoch, logs)
                    )
                )

                self.model.fit(
                    x_train,
                    y_train,
                    epochs=epochs,
                    batch_size=batch_size,
                    validation_data=val_data,
                    callbacks=[model_checkpoint_callback, logging_callback],
                    verbose=0,
                )
        else:
            self.model.fit(
                x_train,
                y_train,
                epochs=epochs,
                batch_size=batch_size,
                validation_data=val_data,
                callbacks=[model_checkpoint_callback],
                verbose=0,
            )



[docs]    def predict(self, x, log_filepath=None):
        if log_filepath is not None:
            self.scores = ["loss"]
            if "metrics" in self.compile_kwargs:
                metric = self.compile_kwargs["metrics"][0]
                self.scores.append(metric)

            str_epoch = (
                lambda epoch, logs: f"epoch: {epoch}, "
                + ", ".join(f"{mtrc}: {logs[mtrc]:.4f}" for mtrc in self.scores)
                + "\n"
            )

            with open(log_filepath, mode="a", buffering=1) as txt_log:
                logging_callback = callbacks.LambdaCallback(
                    on_epoch_end=lambda epoch, logs: txt_log.write(
                        str_epoch(epoch, logs)
                    )
                )
                predictions = self.model.predict(x, callbacks=[logging_callback])

            return predictions
        else:
            return self.model.predict(x)
